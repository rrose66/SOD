Attribute VB_Name = "BAS_RE_ESTABLISH_BASELINES"
Option Compare Database

Sub RE_ESTABLISH_BASELINES()
    Dim STR_SQL As String
'    DoCmd.OpenQuery ("RE_ESTABLISH_BASELINES_01")
    CurrentDb.Execute ("DELETE FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B1.* FROM FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B1")
'    DoCmd.OpenQuery ("RE_ESTABLISH_BASELINES_02")
    STR_SQL = ""
    STR_SQL = STR_SQL & "INSERT INTO FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B1 ( SCANNED, CREDENTIAL, ENVIRONMENT, CONFIG, EXT_SYSTEM, EXT_SERVER, EXT_DB )"
    STR_SQL = STR_SQL & " SELECT FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.SCANNED, FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.CREDENTIAL, FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.ENVIRONMENT, FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.CONFIG, FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.EXT_SYSTEM, FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.EXT_SERVER, FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.EXT_DB"
    STR_SQL = STR_SQL & " FROM FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2"
    CurrentDb.Execute (STR_SQL)
'    DoCmd.OpenQuery ("RE_ESTABLISH_BASELINES_03")
    CurrentDb.Execute ("DELETE FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2.* FROM FEDEBOM_DASHBOARD_EXTERNAL_SYSTEMS_B2")
End Sub
